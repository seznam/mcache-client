83  servers.push_back("server1:11211");
84  servers.push_back("server2:11211");
85  servers.push_back("server3:11211");
86  consistent_hashing_pool_t pool(servers, cfg);
87  // checks if we go through all ring nodes then pool throws
88  std::size_t cnt = 0;
89  std::size_t cnt1 = 0;
90  std::size_t cnt2 = 0;
91  std::size_t cnt3 = 0;
92  // checks whether servers distribution over the ring
93  // and whether we const_iterator walks through all nodes double times
94  std::for_each(pool.begin(), pool.end(), ++var(cnt));
95  // checks whether distribution ring contains node with zero index 40x
96  std::for_each(pool.begin(), pool.end(), if_then(_1 == 0, ++var(cnt1)));
97  // dtto
98  std::for_each(pool.begin(), pool.end(), if_then(_1 == 1, ++var(cnt2)));
99  // dtto
100  std::for_each(pool.begin(), pool.end(), if_then(_1 == 2, ++var(cnt3)));
101  return (cnt == 120) && (cnt1 == 40) && (cnt2 == 40) && (cnt3 == 40);
102 }

110  servers.push_back("server2:11211");

112  fake_consistent_hashing_pool_t pool(servers, cfg);
113  // checks whether key distribution with fake hash functor is like:
114  // [1000] -> 0
115  // [2000] -> 0
116  // [3000] -> 0
117  // [10000] -> 1
118  // [20000] -> 1
119  // [30000] -> 1
120  // [100000] -> 2
121  // [200000] -> 2
122  // [300000] -> 2
123  fake_consistent_hashing_pool_t::const_iterator iidx = pool.choose("a");
124  if ((*iidx != 0) || (*++iidx != 0) || (*++iidx != 0)) return false;
125  iidx = pool.choose("b");
126  if ((*iidx != 1) || (*++iidx != 1) || (*++iidx != 1)) return false;
127  iidx = pool.choose("c");
128  if ((*iidx != 2) || (*++iidx != 2) || (*++iidx != 2)) return false;
129  return (*++iidx == 0) && (*pool.choose("server1:11211") == 0);
